379  template<typename CameraType, typename PixelType>
380  bool readCameraImage(const std::string& cameraName,
381  std::unordered_map<std::string, CameraType*>& interfaceMap,
382  StampedYARPImage<PixelType>& image)
383  {
384  bool ok{true};
385  constexpr std::string_view logPrefix = "[YarpCameraBridge::Impl::readCameraImage] ";
386 
387  if (!checkSensor(interfaceMap, cameraName))
388  {
389  return false;
390  }
391 
392  auto iter = interfaceMap.find(cameraName);
393  auto interface = iter->second;
394 
395  // StampedYARPImage is defined as a pair of yarp::sig::image and double
396  yarp::os::Stamp* txTimestamp{nullptr};
397  if constexpr (std::is_same_v<CameraType, yarp::dev::IFrameGrabberImage>)
398  {
399  if constexpr (!std::is_same_v<PixelType, yarp::sig::PixelRgb>) // (imageType != "RGB")
400  {
401  log()->error("{} Frame Grabber {} handles only RGB image.", logPrefix, cameraName);
402  return false;
403  }
404  ok = interface->getImage(image.first);
405  }
406  else if constexpr (std::is_same_v<CameraType, yarp::dev::IRGBDSensor>)
407  {
408  if constexpr (std::is_same_v<PixelType, yarp::sig::PixelFloat>)// (imageType == "DEPTH")
409  {
410  ok = interface->getDepthImage(image.first, txTimestamp);
411  }
412  }
413 
414  if (!ok)
415  {
416  log()->error("{} Unable to read from {}, use previous image.", logPrefix, cameraName);
417  return false;
418  }
419 
420  image.second = yarp::os::Time::now();
421  return true;
422  }

581  colorImg = cv::Mat(yarpImage.height(), yarpImage.width(), yarp::cv::type_code<yarp::sig::PixelRgb>::value,
