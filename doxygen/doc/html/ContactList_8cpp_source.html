173 bool ContactList::editContact(ContactList::const_iterator element, const PlannedContact& newContact)
174 {
175  if (element == end())
176  {
177  log()->error("[ContactList::addContact] The element is not valid.");
178  return false;
179  }
180 
181  ContactList::const_iterator previousElement = element;
182  ContactList::const_iterator nextElement = element;
183  nextElement++;
184  if (element != begin())
185  {
186  --previousElement;
187  if (newContact.activationTime < previousElement->deactivationTime)

189  log()->error("[ContactList::addContact] The new contact cannot have an activation time "

192  }
193  }
194 
195  if (nextElement != end())
196  {

201  return false;
202  }
203  }
204 
205  m_contacts.erase(element);
206  m_contacts.insert(nextElement, newContact);

211 ContactList::const_iterator ContactList::getPresentContact(double time) const
212 {
213  // With the reverse iterator we find the last step such that the activation time is smaller that time
214  ContactList::const_reverse_iterator presentReverse = std::find_if(rbegin(), rend(),
215  [time](const PlannedContact & a) -> bool { return a.activationTime <= time; });
216  if (presentReverse == rend())
217  {
218  // No contact has activation time lower than the specified time.
219  return end();
220  }
221 
222  return --(presentReverse.base()); //This is to convert a reverse iterator to a forward iterator. The -- is because base() returns a forward iterator to the next element.
223 }
