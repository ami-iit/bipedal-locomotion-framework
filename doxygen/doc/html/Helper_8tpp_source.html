175 template <typename T> void mergeSigVector(yarp::sig::Vector& vector, const T& t)
176 {
177  if constexpr (std::is_arithmetic<T>::value)
178  vector.push_back(t);
179 
180  else if constexpr (!std::is_arithmetic<T>::value)

182  using elementType = typename std::pointer_traits<decltype(t.data())>::element_type;
183  static_assert(std::is_convertible<elementType, double>::value,
184  "[BipedalLocomotion::YarpUtilities::mergeSigVector] The element "
185  "contained in the vector cannot be converted in a double");
186 
187  if constexpr (is_iterable<T>::value)
188  {
189  for (auto it = t.begin(); it != t.end(); it++)
190  vector.push_back(*it);
191 
192  } else if constexpr (has_square_bracket_operator<T>::value)
193  {
194  for (int i = 0; i < t.size(); i++)
195  vector.push_back(t[i]);
196 
197  } else
198  static_assert(dependent_false<T>::value,
199  "[BipedalLocomotion::YarpUtilities::mergeSigVector] The "

203  }

207  static_assert(dependent_false<T>::value,
208  "[BipedalLocomotion::YarpUtilities::mergeSigVector] The type of "
209  "the input element cannot be handled by the function");
210  }

212  return;
