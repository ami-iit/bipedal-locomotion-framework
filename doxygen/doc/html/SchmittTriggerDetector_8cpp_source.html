321 void SchmittTriggerUnit::update(const double& currentTime, const double& rawValue)
322 {
323  if (previousTime == initialTime)
324  {
325  (currentTime > initialTime) ? previousTime = initialTime : previousTime = currentTime;
326  }
327 
328  if (!state)
329  {
330  // Check for transition from false to true - if valid over a timeframe, then switch
331  if (rawValue >= params.onThreshold)
332  {
333  (timer >= params.switchOnAfter) ? state = true : timer += (currentTime - previousTime);
334  if (state)
335  {
336  switchTime = currentTime;
337  timer = 0;
338  }
339  }
340  else
341  {
342  timer = 0;
343  }
344  }
345  else
346  {
347  // check for transition from true to false - if valid over a timeframe, then switch
348  if (rawValue <= params.offThreshold)
349  {
350  (timer >= params.switchOffAfter) ? state = false : timer += (currentTime - previousTime);
351  if (!state)
