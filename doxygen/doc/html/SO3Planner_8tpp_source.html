59 template <class Derived>
60 bool SO3Planner<trivialization>::evaluatePoint(const double& time,
61  manif::SO3d& rotation,
62  manif::SO3TangentBase<Derived>& velocity,
63  manif::SO3TangentBase<Derived>& acceleration) const
64 {
65  if (time < 0 || time > m_T)
66  {
67  std ::cerr << "[SO3Planner::evaluatePoint] The time has to be a real positive number "
68  "between 0 and "
69  << m_T << "." << std::endl;
70  return false;
71  }
72 
73  const double& t = time;
74  // Compute the displacement in the tangent space
75  // - displacement_tangent = log(finalRotation * initialRotation.inverse()) * s(t) (in case of
76  // Right
77  // Trivialization)
78  // - displacement_tangent = log(initialRotation.inverse() * finalRotation) * s(t) (in case of
79  // Left
80  // Trivialization)
81  // You can find the computation of s in "Modern Robotics: Mechanics, Planning, and Control"
82  // (Chapter 9.2)
83  const double s = 10 * std::pow(t / m_T, 3)
84  - 15 * std::pow(t / m_T, 4)
85  + 6 * std::pow(t / m_T, 5);
86  const manif::SO3d::Tangent displacementTangent = m_distance * s;
87 
88  if constexpr (trivialization == LieGroupTrivialization::Right)
89  {
90  // please read this as R(t) = exp(displacementTangent) * initialRotation
91  rotation = displacementTangent.exp() * m_initialRotation;
92  } else
93  // Please read it as trivialization == LieGroupTrivialization::Left
94  {
95  // please read this as R(t) = initialRotation * exp(displacementTangent)
96  rotation = m_initialRotation * displacementTangent.exp();
97  }
98 
99  // compute velocity (it is expressed in body / inertial frame accordingly to the chosen
100  // trivialization)
101  // You can find the computation of sDot in "Modern Robotics: Mechanics, Planning, and Control"
102  // (Chapter 9.2)
103  const double sDot = 10 * 3 * std::pow(t, 2) / std::pow(m_T, 3)
104  - 15 * 4 * std::pow(t, 3) / std::pow(m_T, 4)
105  + 6 * 5 * std::pow(t, 4) / std::pow(m_T, 5);
106  velocity = m_distance * sDot;
107 
108  // compute acceleration (it is expressed in body / inertial frame accordingly to the chosen
109  // trivialization)
110  // You can find the computation of sDdot in "Modern Robotics: Mechanics,
111  // Planning, and Control" (Chapter 9.2)
112  const double sDdot = 10 * 3 * 2 * t / std::pow(m_T, 3)
113  - 15 * 4 * 3 * std::pow(t, 2) / std::pow(m_T, 4)
114  + 6 * 5 * 4 * std::pow(t, 3) / std::pow(m_T, 5);
115  acceleration = m_distance * sDdot;
116 
117  return true;
118 }
