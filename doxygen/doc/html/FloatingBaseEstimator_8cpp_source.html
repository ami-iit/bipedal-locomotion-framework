106 bool FloatingBaseEstimator::advance()
107 {
108  if (m_estimatorState != State::Initialized && m_estimatorState != State::Running)
109  {
110  log()->error("[FloatingBaseEstimator::advance] "
111  "Please initialize the estimator before calling advance().");
112  return false;
113  }
114 
115  bool ok{true};
116  if (m_estimatorState == State::Initialized)
117  {
118  m_estimatorState = State::Running;
119  m_measPrev = m_meas;
120  }
121 
122  if (m_useModelInfo && m_options.kinematicsUpdateEnabled)
123  {
124  // update robot state with previous state estimates
125  if ( (m_meas.encoders.size() == m_modelComp.nrJoints()) ||
126  (m_meas.encodersSpeed.size() == m_modelComp.nrJoints()) )
127  {
128  if (!m_modelComp.kinDyn()->setRobotState(m_estimatorOut.basePose.transform(),
129  m_meas.encoders,
130  m_estimatorOut.baseTwist,
131  m_meas.encodersSpeed,
132  m_options.accelerationDueToGravity))
133  {
134  log()->error("[FloatingBaseEstimator::advance] "
135  "Failed to set kindyncomputations robot state.");
136  return false;
137  }
138  }
139  }
140 
141  ok = ok && predictState(m_measPrev, m_dt);
142  if (m_options.ekfUpdateEnabled)
143  {
144  if (m_useModelInfo && m_options.kinematicsUpdateEnabled)
145  {
146  ok = ok && updateKinematics(m_meas, m_dt);
147  }
148 
149  if (m_options.staticLandmarksUpdateEnabled)
150  {
151  ok = ok && updateLandmarkRelativePoses(m_meas, m_dt);
152  }
153  }
154 
155  ok = ok && updateBaseStateFromIMUState(m_state, m_measPrev,
156  m_estimatorOut.basePose, m_estimatorOut.baseTwist);
157 
158  m_statePrev = m_state;
159  m_measPrev = m_meas;
160 
161  m_estimatorOut.state = m_state;
162  m_estimatorOut.stateStdDev = m_stateStdDev;
163 
164  // clear the map of landmark and contact measurements
165  if (m_useModelInfo)
166  {
167  m_meas.stampedContactsStatus.clear();
168  }
169 
170  if (m_options.staticLandmarksUpdateEnabled)
171  {
172  m_meas.stampedRelLandmarkPoses.clear();
173  }
174  return ok;

795 bool FloatingBaseEstimator::updateBaseStateFromIMUState(const FloatingBaseEstimators::InternalState& state,
796  const FloatingBaseEstimators::Measurements& meas,
797  manif::SE3d& basePose,
798  Eigen::Ref<Eigen::Matrix<double, 6, 1>> baseTwist)
799 {
800  manif::SE3d A_H_IMU(state.imuPosition, state.imuOrientation);
801  Eigen::Matrix<double, 6, 1> v_IMU;
802 
803  v_IMU.head<3>() = state.imuLinearVelocity;
804  if (m_useIMUForAngVelEstimate)
805  {
806  v_IMU.tail<3>() = state.imuOrientation.toRotationMatrix()*(meas.gyro - state.gyroscopeBias);
807  }
808  else
809  {
810  v_IMU.tail<3>() = state.imuAngularVelocity;
811  }
812 
813  Eigen::Matrix<double, 6, 1> tempTwist;
814  if (m_useModelInfo)
815  {

817  {
818  log()->error("[FloatingBaseEstimator::updateBaseStateFromIMUState] "
819  "Failed to get base link state from IMU state.");
820  return false;
821  }
822 
823  if (m_useIMUVelForBaseVelComputation)
824  {
825  baseTwist = tempTwist;
826  }
827  }
828  else
829  {
830  basePose = A_H_IMU;
831  baseTwist = v_IMU;
832  }
833 
834  return true;
