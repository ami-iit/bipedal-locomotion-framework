643 template<typename Class>
644 typename Vector<typename container_data<Class>::type>::resize_function_type DefaultVectorResizer(Class& input)
645 {
646  static_assert (is_resizable<Class>::value, "Class type is not resizable.");
647  static_assert (is_span_constructible<Class>::value || (is_data_available<Class>::value && is_size_available<Class>::value),
648  "Cannot create a span given the provided class.");
649 
650  using value_type = typename container_data<Class>::type;
651 
652  if constexpr (is_span_constructible<Class>::value)
653  {
654  using index_type = typename Vector<value_type>::index_type;
655  using resize_function = typename Vector<value_type>::resize_function_type;
656 
657  Class* inputPtr = &input;
658  resize_function resizeLambda =
659  [inputPtr](index_type newSize) -> iDynTree::Span<value_type>
660  {
661  inputPtr->resize(newSize);
662  return iDynTree::make_span(*inputPtr);
663  };
664 
665  return resizeLambda;
666  }
667  else
668  {
669  using index_type = decltype(std::declval<Class>().size());
670  using resize_function = typename Vector<value_type>::resize_function_type;
671 
672  Class* inputPtr = &input;
673  resize_function resizeLambda =
674  [inputPtr](index_type newSize) -> iDynTree::Span<value_type>
675  {
676  inputPtr->resize(newSize);
677  return iDynTree::make_span(inputPtr->data(), inputPtr->size());
678  };
679 
680  return resizeLambda;
681  }
682 }

693 template<typename Class>
694 Vector<typename container_data<Class>::type>
695 make_vector(Class& input, VectorResizeMode mode = VectorResizeMode::Fixed)
696 {
697  static_assert (!std::is_same<typename container_data<Class>::type, bool>::value,
698  "Cannot create a Vector of bool type. Memory is not contiguos." );
699  static_assert (is_span_constructible<Class>::value || (is_data_available<Class>::value && is_size_available<Class>::value),
700  "Cannot create a span given the provided class.");
701 
702  using value_type = typename container_data<Class>::type;
703 
704  iDynTree::Span<value_type> span;
705 
706  if constexpr (is_span_constructible<Class>::value)
707  {
708  span = iDynTree::make_span(input);
709  }
710  else
711  {
712  span = iDynTree::make_span(input.data(), input.size());
713  }
714 
715  if constexpr (is_resizable<Class>::value)
716  {
717  if (mode == VectorResizeMode::Resizable)
718  {
719  return Vector(span, DefaultVectorResizer(input));
720  }
721  else
722  {
723  return Vector(span);
724  }
725  }
726  else
727  {
728  if (mode == VectorResizeMode::Resizable)
729  {
730  std::cerr << "[GenericContainer::Vector] " << type_name<Class>()
731  << " is not resizable. Returning a non-resizable container." << std::endl;
732  }
733 
734  return Vector(span);
735  }
