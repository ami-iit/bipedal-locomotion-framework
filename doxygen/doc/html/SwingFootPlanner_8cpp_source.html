111 bool SwingFootPlanner::updateSE3Traj()
112 {
113  // compute the trajectory at the current time
114  const double shiftedTime = m_currentTrajectoryTime - m_currentContactPtr->deactivationTime;
115 
116  manif::SO3d rotation;
117 
118  // note here we assume that the velocity is expressed using the mixed representation.
119  // for this reason the velocity can be computed splitting the linear and the angular problem.
120  // For the angular part we are interested in the angular velocity expressed in the inertial
121  // frame (right trivialized velocity). For the liner part we are interested on the derivative of
122  // the position vector w.r.t the time. A similar consideration can be done also for the
123  // acceleration.
124 
125  // Note that here we use asSO3() to access to the angular velocity stored in the 6d-vector. (A
126  // similar consideration can be done also for the acceleration.)
127  auto angularVelocity(m_state.mixedVelocity.asSO3());
128  auto angularAcceleration(m_state.mixedAcceleration.asSO3());
129  if (!m_SO3Planner.evaluatePoint(shiftedTime, rotation, angularVelocity, angularAcceleration))
130  {
131  std::cerr << "[SwingFootPlanner::advance] Unable to get the SO(3) trajectory at time t = "
132  << m_currentTrajectoryTime << "." << std::endl;
133  return false;
134  }
135 
136  manif::SE3d::Translation position;
137  if (!m_planarPlanner.evaluatePoint(m_currentTrajectoryTime,
138  position.head<2>(),
139  m_state.mixedVelocity.coeffs().head<2>(),
140  m_state.mixedAcceleration.coeffs().head<2>()))
141  {
142  std::cerr << "[SwingFootPlanner::advance] Unable to get the planar trajectory at time t = "
143  << m_currentTrajectoryTime << "." << std::endl;
144  return false;
145  }
146 
147  if (!m_heightPlanner.evaluatePoint(m_currentTrajectoryTime,
148  position.segment<1>(2),
149  m_state.mixedVelocity.coeffs().segment<1>(2),
150  m_state.mixedAcceleration.coeffs().segment<1>(2)))
151  {
152  std::cerr << "[SwingFootPlanner::advance] Unable to get the height trajectory at time t = "
153  << m_currentTrajectoryTime << "." << std::endl;
154  return false;
155  }
156 
157  // set the transform
158  m_state.transform = manif::SE3d(position, rotation);
159 
160  m_state.isInContact = false;
161 
162  return true;
163 }

165 bool SwingFootPlanner::advance()
166 {
167  // update the time taking into account the limits
168  m_currentTrajectoryTime
169  = std::min(m_dT + m_currentTrajectoryTime, m_contactList.lastContact()->deactivationTime);
170 
171  // here there are several case
172  // 1. the link was already in contact with the environment and now it is still in contact
173  // 2. the link was in contact with the environment but now not (This cannot happen for the
174  // latest contact phase)
175  // 3. the link was not in contact with the environment and now it is still not in contact (This
176  // cannot happen for the latest contact phase)
177  // 4. the link was not in contact with the environment but now it is in contact (This cannot
178  // happen for the latest contact phase)
179 
180  // This is the case (1). In this case we do not have to update the state
181  if (m_state.isInContact && m_currentTrajectoryTime <= m_currentContactPtr->deactivationTime)
182  {
183  return true;
184  }
185 
186  // In theory this should never happen. This check is only for spotting possible bugs
187  if (m_currentContactPtr == m_contactList.end())
188  {
189  std::cerr << "[SwingFootPlanner::advance] During the last phase the link should be in "
190  "contact with the environment."
191  << std::endl;
192  return false;
193  }
194 
195  // This is the case (2)
196  if (m_state.isInContact && m_currentTrajectoryTime > m_currentContactPtr->deactivationTime)
197  {
198  // create a new trajectory in SE(3)
199  const auto nextContactPtr = std::next(m_currentContactPtr, 1);
200  const double T = nextContactPtr->activationTime - m_currentContactPtr->deactivationTime;
201 
202  if (!m_SO3Planner.setRotations(m_currentContactPtr->pose.asSO3(), nextContactPtr->pose.asSO3(), T))
203  {
204  std::cerr << "[SwingFootPlanner::advance] Unable to set the initial and final "
205  "rotations for the SO(3) planner."
206  << std::endl;
207  return false;
208  }
209 
210  if (!m_planarPlanner.setKnots({m_currentContactPtr->pose.translation().head<2>(),
211  nextContactPtr->pose.translation().head<2>()},
212  {m_currentContactPtr->deactivationTime,
213  nextContactPtr->activationTime}))
214  {
215  std::cerr << "[SwingFootPlanner::advance] Unable to set the knots for the planar "
216  "planner."
217  << std::endl;
218  return false;
219  }
220 
221  const double footHeightViaPointPos = (m_currentContactPtr->pose.translation()(2) +
222  nextContactPtr->pose.translation()(2)) / 2.0 + m_stepHeight;
223 
224  const double footHeightViaPointTime = m_footApexTime * T + m_currentContactPtr->deactivationTime;
225 
226  if (!m_heightPlanner.setKnots({m_currentContactPtr->pose.translation().tail<1>(),
227  Vector1d::Constant(footHeightViaPointPos),
228  nextContactPtr->pose.translation().tail<1>()},
229  {m_currentContactPtr->deactivationTime,
230  footHeightViaPointTime,
231  nextContactPtr->activationTime}))
232  {
233  std::cerr << "[SwingFootPlanner::advance] Unable to set the knots for the height "
234  "planner."
235  << std::endl;
236  return false;
237  }
238 
239  if (!this->updateSE3Traj())
240  {
241  std::cerr << "[SwingFootPlanner::advance] Unable to update the SE(3) trajectory."
242  << std::endl;
243  return false;
244  }
245 
246  return true;
247  }
248 
249 
250  // This is the case (3)
251  if (!m_state.isInContact && m_currentTrajectoryTime > m_currentContactPtr->deactivationTime
252  && m_currentTrajectoryTime <= std::next(m_currentContactPtr, 1)->activationTime)
253  {
254  if (!this->updateSE3Traj())
255  {
256  std::cerr << "[SwingFootPlanner::advance] Unable to update the SE(3) trajectory."
257  << std::endl;
258  return false;
259  }
260 
261 
262  return true;
263  }
264 
265  // This is the case (4)
266  if (!m_state.isInContact && m_currentTrajectoryTime > m_currentContactPtr->deactivationTime
267  && m_currentTrajectoryTime > std::next(m_currentContactPtr, 1)->activationTime)
268  {
269  std::advance(m_currentContactPtr, 1);
270 
271  m_state.transform = m_currentContactPtr->pose;
272  m_state.mixedVelocity.setZero();
273  m_state.mixedAcceleration.setZero();
274  m_state.isInContact = true;
275 
276  return true;
277  }
278 
279  return true;
