567  Eigen::Matrix3d R_pred = R*(omega_skew_dt.exp().rotation());

571  X.imuPosition = p + (v*dt) + (acc*(0.5*dt*dt));
572 
573  if (!meas.lfInContact)
574  {
575  X.lContactFramePosition = X.imuPosition + (R*IMU_H_LF.translation());
576  }
577 
578  if (!meas.rfInContact)

586 bool InvariantEKFBaseEstimator::Impl::updateStates(const Eigen::VectorXd& obs,
587  const Eigen::MatrixXd measModelJacobian,
588  const Eigen::MatrixXd& measNoiseVar,
589  const Eigen::MatrixXd& auxMat,
590  FloatingBaseEstimators::InternalState& state,
591  Eigen::MatrixXd& P)
592 {
593  if (measModelJacobian.cols() != P.rows())
594  {
595  std::cerr << "[InvariantEKFBaseEstimator::updateStates] Measurement model Jacobian size mismatch" << std::endl;
596  return false;
597  }
598 
599  if (measModelJacobian.rows() != measNoiseVar.rows())
600  {
601  std::cerr << "[InvariantEKFBaseEstimator::updateStates] Measurement noise covariance matrix size mismatch" << std::endl;
602  return false;
603  }
604 
605  bool estimateBias;
606  if (P.rows() == m_vecSizeWBias)
607  {
608  estimateBias = true;
609  }
610  else
611  {
612  estimateBias = false;
613  }
614 
615  m_PHT = P*measModelJacobian.transpose();
616  m_S = measModelJacobian*m_PHT + measNoiseVar;
617  m_K = m_PHT*(m_S.inverse());
618 
619  // compute innovation delta
620  extractState(state, m_X, m_theta);
621 
622  int copyTimes = obs.rows()/m_X.cols();
623  if (!copyDiagX(m_X, copyTimes, m_BigX))
624  {
625  return false;
626  }
627 
628  m_innovation = m_BigX*obs; // minus bias terms which are zero in reduced form (Section III C of the paper)
629  m_delta = m_K*auxMat*m_innovation;
630 
631  m_deltaTheta.setZero();
632  if (estimateBias)
633  {
634  m_deltaTheta = m_delta.segment<6>(m_vecSizeWOBias);
635  }
636 
637  // update state
638  if (!calcExpHatX(m_delta.segment(0, m_vecSizeWOBias), m_dX))
639  {
640  std::cerr << "[InvariantEKFBaseEstimator::updateStates] Could not compute state update";
641  return false;
642  }
643 
644  if (!constructState( m_dX*m_X, m_theta+m_deltaTheta, state) ) // right invariant update
645  {
646  std::cerr << "[InvariantEKFBaseEstimator::updateStates] Could not update state";
647  return false;
648  }
649  // update covariance
650  m_IminusKH = Eigen::MatrixXd::Identity(P.rows(), P.cols()) - m_K*measModelJacobian;
651  P = m_IminusKH*P*(m_IminusKH.transpose()) + m_K*measNoiseVar*(m_K.transpose());
652  return true;
653 }

749 bool InvariantEKFBaseEstimator::Impl::calcExpHatX(const Eigen::VectorXd& vec,
750  Eigen::MatrixXd& X)
751 {
752  // Exp(vec) = Exp([ w]) = |ExpSO3(w) JlSO3(w)v JlSO3(w)p JlSO3(w)pr JlSO3(w)pl|
753  // ([ a]) | 1 |
754  // ([ v]) | 1 |
755  // ([vr]) | 1 |
756  // ([vl]) | 1|
757  // where JlSO3 is the left Jacobian of SO(3)
758  if (vec.size() != 15)
759  {
760  std::cerr << "[InvariantEKFBaseEstimator::calcExpHatX] State vector does not seem to have expected size of 15x1." << std::endl;
761  return false;
762  }
763 
764  X.resize(7, 7);
765  auto omega = manif::SO3Tangentd(vec.segment<3>(m_vecOffsets.imuOrientation));
766  X.topLeftCorner<3, 3>() = omega.exp().rotation();
767  X.block<3, 1>(0, 3) = omega.ljac()*vec.segment<3>(m_vecOffsets.imuLinearVel);
768  X.block<3, 1>(0, 4) = omega.ljac()*vec.segment<3>(m_vecOffsets.imuPosition);
769  X.block<3, 1>(0, 5) = omega.ljac()*vec.segment<3>(m_vecOffsets.rContactFramePosition);
770  X.block<3, 1>(0, 6) = omega.ljac()*vec.segment<3>(m_vecOffsets.lContactFramePosition);
771 
772  X.bottomLeftCorner<4, 3>().setZero();
773  X.bottomRightCorner<4, 4>() = Eigen::Matrix4d::Identity();
774  return true;
775 }
