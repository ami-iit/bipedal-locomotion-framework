417  bool setReferences(Eigen::Ref<const Eigen::VectorXd> jointValues)
418  {
419  constexpr auto errorPrefix = "[YarpRobotControl::Impl::setReferences]";
420 
421  if(!this->getJointPos())
422  {
423  log()->error("{} Unable to get the joint position.", errorPrefix);
424  return false;
425  }
426 
427  const auto worstError = this->getWorstPositionDirectError(jointValues,
428  this->positionFeedback);
429 
430  if (worstError.second > this->positionDirectMaxAdmissibleError)
431  {
432  log()->error("{} The worst error between the current and the desired position of the "
433  "joint named '{}' is greater than {} deg. Error = {} deg.",
434  errorPrefix,
435  this->axesName[worstError.first],
436  180 / M_PI * this->positionDirectMaxAdmissibleError,
437  180 / M_PI * worstError.second);
438  return false;
439  }
440 
441  for (const auto& [mode, indeces] : this->desiredJointValuesAndMode.index)
442  {
443  // if indeces vector is empty no joint is controlled with this specific control mode
444  if (indeces.empty())
445  continue;
446 
447  if (mode == IRobotControl::ControlMode::Idle)
448  continue;
449 
450  else if (mode == IRobotControl::ControlMode::Unknown)
451  {
452  std::string joints = "";
453  for (const auto& index : indeces)
454  joints += " '" + this->axesName[index] + "'";
455 
456  log()->error("{} The following joints does not have a specified control "
457  "mode:{}. Please set a feasible control mode.",
458  errorPrefix,
459  joints);
460  return false;
461 
462  } else if (mode == IRobotControl::ControlMode::Position)
463  {
464  for (int i = 0; i < indeces.size(); i++)
465  {
466  const auto jointError = std::abs(jointValues[indeces[i]]
467  - this->positionFeedback[indeces[i]]);
468 
469  constexpr double scaling = 180 / M_PI;
470  constexpr double minVelocityInDegPerSeconds = 3.0;
471  this->positionControlRefSpeeds[i]
472  = std::max(minVelocityInDegPerSeconds,
473  scaling * (jointError / this->positioningDuration));
474  }
475 
476  if (!this->positionInterface->setRefSpeeds(indeces.size(),
477  indeces.data(),
478  this->positionControlRefSpeeds.data()))
479  {
480  log()->error("{} Unable to set the reference speed for the position control "
481  "joints.",
482  errorPrefix);
483  return false;
484  }
485 
486  this->startPositionControlInstant = yarp::os::Time::now();
487  }
488 
489  // Yarp wants the quantities in degrees
490  double scaling = 180 / M_PI;
491  if (mode == ControlMode::Torque)
492  scaling = 1;
493 
494  for (int i = 0; i < indeces.size(); i++)
495  this->desiredJointValuesAndMode.value[mode][i] = scaling * jointValues[indeces[i]];
496 
497  if (!this->control(mode)(indeces.size(),
498  indeces.data(),
499  this->desiredJointValuesAndMode.value[mode].data()))
500 
501  {
502  log()->error("{} Unable to set the desired joint values.", errorPrefix);
503  return false;
504  }
505  }
506  return true;
507  }
